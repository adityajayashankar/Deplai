    <script>
        // ==========================================
        // 1. NODE MAP PHYSICS ENGINE (Upgraded for large canvas)
        // ==========================================
        const canvas = document.getElementById('nodeMap');
        const ctx = canvas.getContext('2d');
        
        let nodes = [];
        let connections = [];
        let mouseX = -1000;
        let mouseY = -1000;

        function initCanvas() {
            const rect = canvas.parentElement.getBoundingClientRect();
            canvas.width = rect.width * 2;
            canvas.height = rect.height * 2;
            ctx.scale(2, 2);
            
            const cw = rect.width;
            const ch = rect.height;

            // Define central Core node
            const centerNode = { x: cw/2, y: ch/2, radius: 32, type: 'center', vx: 0, vy: 0, baseX: cw/2, baseY: ch/2 };
            nodes = [centerNode];
            connections = [];

            // Define Primary child nodes (Inner Ring)
            const numPrimary = 6;
            const primaryDist = 120;
            for(let i=0; i<numPrimary; i++) {
                const angle = (i * 2 * Math.PI) / numPrimary;
                const nx = cw/2 + Math.cos(angle) * primaryDist;
                const ny = ch/2 + Math.sin(angle) * primaryDist;
                nodes.push({ x: nx, y: ny, radius: 12, type: 'child', vx: 0, vy: 0, baseX: nx, baseY: ny, offset: i });
                connections.push({ from: 0, to: nodes.length - 1 });
            }

            // Define Secondary child nodes (Outer Ring)
            const numSecondary = 8;
            const secondaryDist = 240;
            for(let i=0; i<numSecondary; i++) {
                const angle = (i * 2 * Math.PI) / numSecondary + 0.5;
                const nx = cw/2 + Math.cos(angle) * secondaryDist;
                const ny = ch/2 + Math.sin(angle) * secondaryDist;
                nodes.push({ x: nx, y: ny, radius: 8, type: 'secondary', vx: 0, vy: 0, baseX: nx, baseY: ny, offset: i * 2 });
                // Connect to a random primary node
                const randomPrimary = Math.floor(Math.random() * numPrimary) + 1;
                connections.push({ from: randomPrimary, to: nodes.length - 1 });
                // Sometimes connect secondary nodes together
                if (i > 0 && Math.random() > 0.5) {
                    connections.push({ from: nodes.length - 2, to: nodes.length - 1 });
                }
            }
        }

        canvas.addEventListener('mousemove', (e) => {
            const rect = canvas.getBoundingClientRect();
            mouseX = e.clientX - rect.left;
            mouseY = e.clientY - rect.top;
        });
        canvas.addEventListener('mouseleave', () => {
            mouseX = -1000; mouseY = -1000;
        });

        function animateNodes(time) {
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // Update Physics
            nodes.forEach(node => {
                if (node.type === 'child' || node.type === 'secondary') {
                    // Complex sine wave floating based on time, type, and offset
                    const speed = node.type === 'child' ? 1000 : 1500;
                    const magnitude = node.type === 'child' ? 8 : 15;
                    const hoverOffsetX = Math.cos(time / speed + node.offset) * magnitude;
                    const hoverOffsetY = Math.sin(time / speed + node.offset) * magnitude;
                    node.targetX = node.baseX + hoverOffsetX;
                    node.targetY = node.baseY + hoverOffsetY;
                } else {
                    node.targetX = node.baseX;
                    node.targetY = node.baseY;
                }

                // Mouse Repulsion Field
                const dx = mouseX - node.x;
                const dy = mouseY - node.y;
                const dist = Math.sqrt(dx*dx + dy*dy);
                
                const repulseRadius = 100;
                if (dist < repulseRadius) {
                    const force = (repulseRadius - dist) / repulseRadius;
                    node.targetX -= (dx / dist) * force * 30;
                    node.targetY -= (dy / dist) * force * 30;
                }

                // Spring back
                node.x += (node.targetX - node.x) * 0.08;
                node.y += (node.targetY - node.y) * 0.08;
            });

            // Draw Connections
            connections.forEach(conn => {
                const n1 = nodes[conn.from];
                const n2 = nodes[conn.to];
                
                const grad = ctx.createLinearGradient(n1.x, n1.y, n2.x, n2.y);
                grad.addColorStop(0, 'rgba(255, 122, 158, 0.4)');
                grad.addColorStop(1, 'rgba(255, 122, 158, 0.05)');
                
                ctx.beginPath();
                // Make outer connections thinner
                ctx.lineWidth = nodes[conn.to].type === 'secondary' ? 1 : 2;
                ctx.strokeStyle = grad;
                ctx.moveTo(n1.x, n1.y);
                ctx.lineTo(n2.x, n2.y);
                ctx.stroke();

                // Draw moving data packets along primary lines occasionally
                if (nodes[conn.to].type === 'child' && (Math.floor(time / 50) + conn.to) % 30 === 0) {
                     ctx.beginPath();
                     ctx.fillStyle = '#FF7A9E';
                     const progress = (time % 1000) / 1000;
                     const px = n1.x + (n2.x - n1.x) * progress;
                     const py = n1.y + (n2.y - n1.y) * progress;
                     ctx.arc(px, py, 2, 0, Math.PI * 2);
                     ctx.fill();
                     ctx.shadowBlur = 10;
                     ctx.shadowColor = '#FF7A9E';
                     ctx.fill();
                     ctx.shadowBlur = 0;
                }
            });

            // Draw Nodes
            nodes.forEach(node => {
                ctx.beginPath();
                ctx.arc(node.x, node.y, node.radius, 0, Math.PI * 2);
                
                if (node.type === 'center') {
                    ctx.fillStyle = '#050505';
                    ctx.fill();
                    ctx.lineWidth = 4;
                    ctx.strokeStyle = '#FF7A9E';
                    
                    // Core pulsing effect
                    const pulse = Math.sin(time / 300) * 5 + 15;
                    ctx.shadowBlur = pulse;
                    ctx.shadowColor = 'rgba(255, 122, 158, 0.6)';
                    ctx.stroke();
                    ctx.shadowBlur = 0; 
                } else if (node.type === 'child') {
                    ctx.fillStyle = '#1A1418';
                    ctx.fill();
                    ctx.lineWidth = 2;
                    ctx.strokeStyle = '#FF7A9E';
                    ctx.stroke();
                } else {
                    ctx.fillStyle = 'rgba(255, 122, 158, 0.1)';
                    ctx.fill();
                    ctx.lineWidth = 1;
                    ctx.strokeStyle = 'rgba(255, 122, 158, 0.5)';
                    ctx.stroke();
                }
            });

            requestAnimationFrame(animateNodes);
        }

        // Initialize and handle resize
        initCanvas();
        requestAnimationFrame(animateNodes);
        window.addEventListener('resize', initCanvas);


        // ==========================================
        // 3. TELEMETRY DASHBOARD (Now HUD Overlay)
        // ==========================================
        
        const circle = document.getElementById('telemetry-ring');
        const valText = document.getElementById('telemetry-val');
        const targetPercent = 92;
        const circumference = 2 * Math.PI * 45; // r=45
        
        // Initial setup
        circle.style.strokeDasharray = `${circumference} ${circumference}`;
        circle.style.strokeDashoffset = circumference;

        setTimeout(() => {
            const offset = circumference - (targetPercent / 100) * circumference;
            circle.style.strokeDashoffset = offset;
            
            // Counter animation
            let current = 0;
            const timer = setInterval(() => {
                current += 2;
                if(current >= targetPercent) {
                    current = targetPercent;
                    clearInterval(timer);
                }
                valText.innerText = current + '%';
            }, 30);
        }, 500);

        // Dynamic Equalizer Bars
        const eqContainer = document.getElementById('equalizer-bars');
        const numBars = 10; // Increased bars for HUD
        const bars = [];

        for(let i=0; i<numBars; i++) {
            const bar = document.createElement('div');
            bar.className = 'flex-1 bg-accent/80 rounded-t-sm transition-all duration-300 shadow-[0_0_5px_var(--accent-glow)]';
            bar.style.height = '10%'; 
            eqContainer.appendChild(bar);
            bars.push(bar);
        }

        setInterval(() => {
            bars.forEach(bar => {
                if (bar === bars[bars.length - 1] || bar === bars[bars.length - 2]) {
                    bar.style.height = (75 + Math.random() * 25) + '%';
                } else {
                    const height = 15 + Math.random() * 65;
                    bar.style.height = height + '%';
                    
                    if(height < 35) bar.style.backgroundColor = 'rgba(255, 122, 158, 0.3)';
                    else bar.style.backgroundColor = '#FF7A9E';
                }
            });
        }, 400);

    </script>
