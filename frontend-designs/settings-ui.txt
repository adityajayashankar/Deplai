// ==========================================
// PIXEL EFFECT FOR BUTTONS
// ==========================================
class Pixel {
  constructor(canvas, context, x, y, color, speed, delay) {
    this.width = canvas.width;
    this.height = canvas.height;
    this.ctx = context;
    this.x = x; this.y = y; this.color = color;
    this.speed = (Math.random() * 0.8 + 0.1) * speed;
    this.size = 0;
    this.sizeStep = Math.random() * 0.4;
    this.minSize = 0.5;
    this.maxSizeInteger = 2;
    this.maxSize = Math.random() * (this.maxSizeInteger - this.minSize) + this.minSize;
    this.delay = delay; this.counter = 0;
    this.counterStep = Math.random() * 4 + (this.width + this.height) * 0.01;
    this.isIdle = false; this.isReverse = false; this.isShimmer = false;
  }
  draw() {
    const o = this.maxSizeInteger * 0.5 - this.size * 0.5;
    this.ctx.fillStyle = this.color;
    // Removed Math.round to allow crisp sub-pixel rendering 
    this.ctx.fillRect(this.x + o, this.y + o, this.size, this.size);
  }
  appear() {
    this.isIdle = false;
    if (this.counter <= this.delay) { this.counter += this.counterStep; return; }
    if (this.size >= this.maxSize) this.isShimmer = true;
    if (this.isShimmer) this.shimmer(); else this.size += this.sizeStep;
    this.draw();
  }
  disappear() {
    this.isShimmer = false; this.counter = 0;
    if (this.size <= 0) { this.isIdle = true; return; }
    else this.size -= 0.1;
    this.draw();
  }
  shimmer() {
    if (this.size >= this.maxSize) this.isReverse = true;
    else if (this.size <= this.minSize) this.isReverse = false;
    if (this.isReverse) this.size -= this.speed; else this.size += this.speed;
  }
}

const PixelButton = ({ children, onClick, disabled = false, variant = 'default', className = '', 'aria-label': ariaLabel }) => {
  const containerRef = useRef(null);
  const canvasRef = useRef(null);
  const pixelsRef = useRef([]);
  const animationRef = useRef(null);
  
  // Use high-contrast pixels based on the button background
  const colors = variant === 'danger' ? '#ffffff,#fca5a5,#b91c1c' : 
                 variant === 'outline' ? '#ffffff,#d4d4d8,#a1a1aa' : 
                 '#000000,#3f3f46,#71717a';

  const initPixels = useCallback(() => {
    if (!containerRef.current || !canvasRef.current) return;
    const rect = containerRef.current.getBoundingClientRect();
    if (rect.width === 0 || rect.height === 0) return;

    const dpr = window.devicePixelRatio || 1;
    canvasRef.current.width = rect.width * dpr;
    canvasRef.current.height = rect.height * dpr;
    const ctx = canvasRef.current.getContext('2d');
    ctx.scale(dpr, dpr);

    const colorsArray = colors.split(',');
    const pxs = [];
    // Spaced tightly for denser effect
    for (let x = 0; x < rect.width; x += 5) {
      for (let y = 0; y < rect.height; y += 5) {
        const color = colorsArray[Math.floor(Math.random() * colorsArray.length)];
        pxs.push(new Pixel(canvasRef.current, ctx, x, y, color, 0.06, 0));
      }
    }
    pixelsRef.current = pxs;
  }, [colors]);

  const doAnimate = useCallback((fnName) => {
    animationRef.current = requestAnimationFrame(() => doAnimate(fnName));
    const ctx = canvasRef.current?.getContext('2d');
    if (!ctx || !canvasRef.current) return;
    
    const dpr = window.devicePixelRatio || 1;
    ctx.clearRect(0, 0, canvasRef.current.width / dpr, canvasRef.current.height / dpr);
    
    let allIdle = true;
    pixelsRef.current.forEach(p => { p[fnName](); if (!p.isIdle) allIdle = false; });
    if (allIdle && animationRef.current) cancelAnimationFrame(animationRef.current);
  }, []);

  useEffect(() => {
    initPixels();
    // Add ResizeObserver so canvas adjusts if tab layout shifts
    const ro = new ResizeObserver(() => initPixels());
    if (containerRef.current) ro.observe(containerRef.current);
    
    return () => {
      ro.disconnect();
      animationRef.current && cancelAnimationFrame(animationRef.current);
    };
  }, [initPixels]);

  const baseStyles = "relative overflow-hidden flex items-center justify-center px-4 py-2 rounded-md font-semibold text-sm transition-colors duration-200 disabled:opacity-50 disabled:cursor-not-allowed";
  const variants = {
    default: `bg-zinc-100 text-black hover:bg-white ${T.border}`,
    danger: `bg-[#EF4444] text-white hover:bg-red-500`,
    outline: `bg-transparent text-zinc-300 border ${T.border} hover:border-zinc-500 hover:text-white`
  };

  return (
    <button
      ref={containerRef}
      onClick={onClick}
      disabled={disabled}
      aria-label={ariaLabel}
      className={`${baseStyles} ${variants[variant]} ${className}`}
      onMouseEnter={() => { if (!disabled && pixelsRef.current.length > 0) { animationRef.current && cancelAnimationFrame(animationRef.current); doAnimate('appear'); } }}
      onMouseLeave={() => { if (!disabled && pixelsRef.current.length > 0) { animationRef.current && cancelAnimationFrame(animationRef.current); doAnimate('disappear'); } }}
    >
      <canvas ref={canvasRef} className="absolute inset-0 pointer-events-none block" />
      <span className="relative z-10 flex items-center gap-2">{children}</span>
    </button>
  );
};

export default function SettingsModule() {
  const [activeContext, setActiveContext] = useState('workspace');
  const [activeTab, setActiveTab] = useState('integrations');
  const [isMobileMenuOpen, setIsMobileMenuOpen] = useState(false);

  const workspaceTabs = [
    { id: 'integrations', label: 'Integrations', icon: Github },
    { id: 'cloud', label: 'Cloud & Deploy', icon: Cloud },
    { id: 'ai', label: 'AI Orchestration', icon: Cpu },
    { id: 'secrets', label: 'Workspace Secrets', icon: Key },
    { id: 'danger', label: 'Danger Zone', icon: AlertTriangle },
  ];

  const userTabs = [
    { id: 'account', label: 'Account', icon: User },
    { id: 'security', label: 'Security', icon: Shield },
    { id: 'notifications', label: 'Notifications', icon: Bell },
    { id: 'preferences', label: 'Preferences', icon: Settings },
    { id: 'aidefaults', label: 'AI Defaults', icon: Bot },
    { id: 'user_integrations', label: 'Integrations', icon: Github },
    { id: 'privacy', label: 'Privacy & Data', icon: Lock },
    { id: 'billing', label: 'Billing & Usage', icon: Activity },
    { id: 'sessions', label: 'Devices & Sessions', icon: Laptop },
  ];

  const allTabs = [...workspaceTabs, ...userTabs];

  useEffect(() => {
    setActiveTab(activeContext === 'workspace' ? 'integrations' : 'account');
  }, [activeContext]);

  // FIX 4: active tab label for mobile header
  const activeLabel = allTabs.find(t => t.id === activeTab)?.label ?? '';

  const SidebarNav = ({ tabs, context, label }) => (
    <div className="mb-8">
      <h4 className={`px-6 text-[10px] font-bold uppercase tracking-widest ${T.hint} mb-3`}>{label}</h4>
      <div className="flex flex-col px-3 space-y-0.5">
        {tabs.map(tab => {
          const Icon = tab.icon;
          const isActive = activeContext === context && activeTab === tab.id;
          return (
            <button
              key={tab.id}
              onClick={() => { setActiveContext(context); setActiveTab(tab.id); setIsMobileMenuOpen(false); }}
              aria-current={isActive ? 'page' : undefined}
              className={`w-full text-left px-3 py-2 text-sm font-medium rounded-md transition-all flex items-center justify-between ${isActive ? 'bg-[#111318] text-white' : 'text-zinc-400 hover:text-white hover:bg-[#111318]/50'}`}
            >
              <span className="flex items-center gap-3"><Icon size={16} aria-hidden="true" />{tab.label}</span>
              {isActive && <div className={`w-1 h-4 rounded-full ${context === 'workspace' ? 'bg-[#4F6EF7]' : 'bg-white'}`} />}
            </button>
          );
        })}
      </div>
    </div>
  );

  return (
    <div className={`min-h-screen ${T.bg} ${T.text} flex font-sans selection:bg-zinc-800`}>
      {/* Mobile overlay */}
      {isMobileMenuOpen && (
        <div className="fixed inset-0 bg-black/60 z-40 md:hidden" onClick={() => setIsMobileMenuOpen(false)} aria-hidden="true" />
      )}

      {/* Sidebar */}
      <nav
        aria-label="Settings navigation"
        className={`w-[240px] shrink-0 border-r ${T.border} flex flex-col h-screen fixed md:sticky top-0 bg-[#0A0B0F] z-50 transition-transform duration-300 ${isMobileMenuOpen ? 'translate-x-0' : '-translate-x-full md:translate-x-0'}`}
      >
        <div className={`h-16 flex items-center justify-between px-6 border-b ${T.border}`}>
          <div className="flex items-center gap-3">
            <div className="w-6 h-6 bg-[#4F6EF7] text-white font-bold flex items-center justify-center text-xs rounded-sm" aria-hidden="true">D</div>
            <span className="font-semibold tracking-wide text-sm">DeplAI</span>
          </div>
          <button className="md:hidden text-zinc-400 hover:text-white" onClick={() => setIsMobileMenuOpen(false)} aria-label="Close menu">
            <X className="w-5 h-5" aria-hidden="true" />
          </button>
        </div>
        <div className="flex-1 overflow-y-auto custom-scrollbar py-6">
          <SidebarNav tabs={workspaceTabs} context="workspace" label="Workspace" />
          <SidebarNav tabs={userTabs} context="user" label="Your Account" />
        </div>
      </nav>

      {/* Main */}
      <div className="flex-1 flex flex-col min-w-0">
        <header className={`h-16 flex items-center gap-3 px-4 sm:px-8 border-b ${T.border} bg-[#0A0B0F]/90 backdrop-blur-sm sticky top-0 z-10`}>
          <button
            onClick={() => setIsMobileMenuOpen(true)}
            className="md:hidden p-1.5 -ml-2 text-zinc-400 hover:text-white"
            aria-label="Open navigation menu"
          >
            <Menu className="w-5 h-5" aria-hidden="true" />
          </button>

          {/* FIX 4: Mobile shows active tab name; desktop shows full breadcrumb */}
          <span className="text-sm font-semibold text-white md:hidden">{activeLabel}</span>
          <div className="hidden md:flex items-center gap-2 text-sm">
            <span className={T.muted}>Settings</span>
            <span className={T.hint}>›</span>
            <span className={T.muted}>{activeContext === 'workspace' ? 'Workspace' : 'Account'}</span>
            <span className={T.hint}>›</span>
            <span className="font-semibold text-white">{activeLabel}</span>
          </div>
        </header>

        <main className="flex-1 overflow-y-auto custom-scrollbar p-4 sm:p-8">
          <div className="max-w-3xl mx-auto w-full">
            {activeContext === 'workspace' && (
              <>
                {activeTab === 'integrations' && <WorkspaceIntegrations />}
                {activeTab === 'cloud' && <WorkspaceCloudDeploy />}
                {activeTab === 'ai' && <WorkspaceAIOrchestration />}
                {activeTab === 'secrets' && <WorkspaceSecrets />}
                {activeTab === 'danger' && <WorkspaceDangerZone />}
              </>
            )}
            {activeContext === 'user' && (
              <>
                {activeTab === 'account' && <UserAccount />}
                {activeTab === 'security' && <UserSecurity />}
                {activeTab === 'notifications' && <UserNotifications />}
                {activeTab === 'preferences' && <UserPreferences />}
                {activeTab === 'aidefaults' && <UserAIDefaults />}
                {activeTab === 'user_integrations' && <UserIntegrations />}
                {activeTab === 'privacy' && <UserPrivacyData />}
                {activeTab === 'billing' && <UserBillingUsage />}
                {activeTab === 'sessions' && <UserDevicesSessions />}
              </>
            )}
          </div>
        </main>
      </div>

      <style dangerouslySetInnerHTML={{__html: `
        .custom-scrollbar::-webkit-scrollbar { width: 6px; }
        .custom-scrollbar::-webkit-scrollbar-track { background: transparent; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background-color: #2D3039; border-radius: 10px; }
        .custom-scrollbar::-webkit-scrollbar-thumb:hover { background-color: #3f3f46; }
        @keyframes fadeIn { from { opacity: 0; transform: translateY(5px); } to { opacity: 1; transform: translateY(0); } }
        .animate-in { animation: fadeIn 0.2s ease-out forwards; }
      `}} />
    </div>
  );
}
